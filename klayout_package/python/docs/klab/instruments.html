 31class KlabInstrument:
 32    """Base class for all klab instruments."""
 33    
 34    def __init__(self, name, address, communication_backend: CommunicationBackend = None, **kwargs):
 35        """Initialize a new instrument.
 36        
 37        Args:
 38            name: A friendly name for the instrument
 39            address: Address string for the instrument
 40            communication_backend: Custom communication backend instance. 
 41                                 If None, defaults to VisaBackend for backward compatibility.
 42            **kwargs: Additional configuration options
 43        """
 44        self.name = name
 45        self.address = address
 46        
 47        # Use provided backend or default to VISA
 48        if communication_backend is not None:
 49            self._comm_backend = communication_backend
 50        else:
 51            self._comm_backend = VisaBackend()
 52        
 53        # Legacy attributes for backward compatibility
 54        self._visa_instrument = None
 55        self._rm = None
 56
 57        # Check environment variable to control data stream printing
 58        self._debug_stream = environ.get('KLAB_DEBUG_STREAM', 'False').lower() in ('true', '1', 't')
 59
 60        # Connect to the instrument if requested
 61        if kwargs.get('connect', True):
 62            self.connect()
 63    
 64
 65    def connect(self):
 66        """Establish a connection to the instrument."""
 67        try:
 68            success = self._comm_backend.connect(self.address)
 69            if success:
 70                print(f"Connected to {self.name} at {self.address}")
 71                # For backward compatibility with VISA instruments
 72                if isinstance(self._comm_backend, VisaBackend):
 73                    self._visa_instrument = self._comm_backend._visa_instrument
 74                    self._rm = self._comm_backend._rm
 75            else:
 76                print(f"Failed to connect to {self.name} at {self.address}")
 77        except Exception as e:
 78            print(f"Failed to connect to {self.name} at {self.address}: {e}")
 79    
 80    def disconnect(self):
 81        """Close the connection to the instrument."""
 82        try:
 83            self._comm_backend.disconnect()
 84            # Clear legacy attributes
 85            self._visa_instrument = None
 86            self._rm = None
 87            print(f"Disconnected from {self.name}")
 88        except Exception as e:
 89            print(f"Error disconnecting from {self.name}: {e}")
 90    
 91    def write(self, command):
 92        """Send a command to the instrument."""
 93        if self._debug_stream:
 94            print(f"\t[{self.name}] > WRITE: {command}")
 95        
 96        try:
 97            self._comm_backend.write(command)
 98        except Exception as e:
 99            print(f"Warning: Failed to write to {self.name}: {e}")
100        return None
101    
102    def read(self):
103        """Read a response from the instrument."""
104        if self._debug_stream:
105            print(f"\t[{self.name}] > READ")
106
107        try:
108            response = self._comm_backend.read()
109            if self._debug_stream:
110                # repr() is used to show hidden characters like newlines
111                print(f"\t[{self.name}] > RECV : {repr(response)}")
112            return response
113        except Exception as e:
114            print(f"Warning: Failed to read from {self.name}: {e}")
115            return None
116    
117    def query(self, command):
118        """Send a query and return the response."""
119        if self._debug_stream:
120            print(f"\t[{self.name}] > QUERY: {command}")
121
122        try:
123            response = self._comm_backend.query(command)
124            if self._debug_stream:
125                # repr() is used to show hidden characters like newlines
126                print(f"\t[{self.name}] > RECV : {repr(response)}")
127            return response
128        except Exception as e:
129            print(f"Warning: Failed to query {self.name}: {e}")
130            return None
131    
132    def close(self):
133        """Alias for disconnect to ensure compatibility with other libraries."""
134        self.disconnect()
135
136    def wait(self, seconds):
137        """Wait for a specified number of seconds."""
138        if self._debug_stream:
139            print(f"\t[{self.name}] > WAIT: {seconds} seconds")
140        
141        import time
142        time.sleep(seconds)
143
144    def is_connected(self):
145        """Check if the instrument is currently connected."""
146        try:
147            return self._comm_backend.is_connected()
148        except Exception:
149            return False
150        
151    def __del__(self):
152        """Ensure the instrument is disconnected when the object is deleted."""
153        try:
154            self.disconnect()
155        except Exception:
156            pass  # Ignore errors during cleanup
157
158    def __repr__(self):
159        """String representation of the instrument."""
160        return f"<KlabInstrument name={self.name}, address={self.address}>"

class VisaBackend(klab.instruments.CommunicationBackend): 
 51class VisaBackend(CommunicationBackend):
 52    """VISA communication backend for SCPI instruments."""
 53    
 54    def __init__(self):
 55        self._visa_instrument = None
 56        self._rm = None
 57    
 58    def connect(self, address: str) -> bool:
 59        """Establish a VISA connection to the instrument."""
 60        try:
 61            self._rm = pyvisa.ResourceManager()
 62            self._visa_instrument = self._rm.open_resource(address)
 63            return True
 64        except Exception as e:
 65            print(f"VISA connection failed: {e}")
 66            self._visa_instrument = None
 67            return False
 68    
 69    def disconnect(self):
 70        """Close the VISA connection."""
 71        if self._visa_instrument:
 72            self._visa_instrument.close()
 73            self._visa_instrument = None
 74        if self._rm:
 75            self._rm.close()
 76            self._rm = None
